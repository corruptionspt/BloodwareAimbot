local dummy, attach0, attach1, alignPos, alignOri
local toggles = {}
local tweenActive = false

local function hum() return plr.Character and plr.Character:FindFirstChild("Humanoid") end

local function makeButton(name, y)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(0, 190, 0, 40)
    btn.Position = UDim2.new(0.5, -95, 0, y)
    btn.BackgroundColor3 = Color3.fromRGB(200, 40, 50)
    btn.Text = name
    btn.TextColor3 = Color3.new(1,1,1)
    btn.Font = Enum.Font.Arcade
    btn.TextSize = 18
    btn.AutoButtonColor = false
    btn.Parent = main
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 10)

    toggles[name] = false

    btn.MouseButton1Click:Connect(function()
        toggles[name] = not toggles[name]
        TweenService:Create(btn, TweenInfo.new(0.22), {
            BackgroundColor3 = toggles[name] and Color3.fromRGB(40, 200, 70) or Color3.fromRGB(200, 40, 50)
        }):Play()

        if name == "Float" then
            if toggles[name] then
                floatPart = Instance.new("Part")
                floatPart.Size = Vector3.new(6,1,6)
                floatPart.Anchored = true
                floatPart.CanCollide = true
                floatPart.Transparency = 0.6
                floatPart.Material = Enum.Material.ForceField
                floatPart.Color = Color3.fromRGB(150,150,255)
                floatPart.Parent = workspace

                task.spawn(function()
                    while toggles["Float"] and floatPart and root() do
                        floatPart.Position = root().Position - Vector3.new(0,3.2,0)
                        task.wait()
                    end
                end)
            else
                if floatPart then floatPart:Destroy() floatPart = nil end
            end

        elseif name == "Small Hitbox" then
            if toggles[name] and root() then
                pcall(function()
                    local char = plr.Character
                    if char:FindFirstChild("Animate") then char.Animate:Destroy() end
                    if char:FindFirstChild("__HITBOX") then
                        char.__HITBOX.Size = Vector3.new(0.5,0.5,0.5)
                    end
                    root().Size = Vector3.new(1,2,1)
                end)
            end

        elseif name == "Tween to Base" then
            if toggles[name] then
                if tweenActive then return end
                tweenActive = true
                task.spawn(function()
                    if not root() or not hum() then tweenActive = false return end

                    hum().PlatformStand = true
                    task.wait(0.2)
                    hum().PlatformStand = false

                    local targetPos = nil
                    for _, plot in pairs(workspace.Plots:GetChildren()) do
                        local sign = plot:FindFirstChild("PlotSign")
                        if sign then
                            local label = sign:FindFirstChild("SurfaceGui")
                            if label then label = label:FindFirstChild("Frame") end
                            if label then label = label:FindFirstChild("TextLabel") end
                            if label and label.Text == plr.DisplayName.."'s Base" then
                                local box = plot:FindFirstChild("DeliveryHitbox")
                                if box then
                                    targetPos = box.Position + Vector3.new(0,6,0)
                                    break
                                end
                            end
                        end
                    end

                    if not targetPos then
                        tweenActive = false
                        toggles["Tween to Base"] = false
                        btn.BackgroundColor3 = Color3.fromRGB(200,40,50)
                        return
                    end

                    dummy = Instance.new("Part")
                    dummy.Anchored = true
                    dummy.Transparency = 1
                    dummy.CanCollide = false
                    dummy.Position = targetPos
                    dummy.Parent = workspace

                    attach0 = Instance.new("Attachment", root())
                    attach1 = Instance.new("Attachment", dummy)

                    alignPos = Instance.new("AlignPosition", root())
                    alignPos.Attachment0 = attach0
                    alignPos.Attachment1 = attach1
                    alignPos.MaxForce = 999999
                    alignPos.MaxVelocity = 28
                    alignPos.Responsiveness = 200

                    alignOri = Instance.new("AlignOrientation", root())
                    alignOri.Attachment0 = attach0
                    alignOri.Attachment1 = attach1
                    alignOri.MaxTorque = 999999
                    alignOri.Responsiveness = 200

                    task.spawn(function()
                        while alignPos and (root().Position - targetPos).Magnitude > 5 do task.wait(0.1) end
                        pcall(function()
                            alignPos:Destroy()
                            alignOri:Destroy()
                            attach0:Destroy()
                            attach1:Destroy()
                            dummy:Destroy()
                        end)
                        task.wait(0.8)
                        if hum() then hum().PlatformStand = false end
                        tweenActive = false
                        toggles["Tween to Base"] = false
                        btn.BackgroundColor3 = Color3.fromRGB(200,40,50)
                    end)
                end)
            else
                tweenActive = false
                pcall(function()
                    if alignPos then alignPos:Destroy() end
                    if alignOri then alignOri:Destroy() end
                    if attach0 then attach0:Destroy() end
                    if attach1 then attach1:Destroy() end
                    if dummy then dummy:Destroy() end
                end)
                if hum() then hum().PlatformStand = false end
            end

        elseif name == "Steal Booster" then
            if hum() then
                hum().JumpPower = toggles[name] and 65 or 50
                hum().JumpHeight = toggles[name] and 25 or 7.2
            end
            if toggles[name] then
                task.spawn(function()
                    while toggles["Steal Booster"] and hum() and root() do
                        local dir = hum().MoveDirection * 28
                        root().Velocity = Vector3.new(dir.X, root().Velocity.Y, dir.Z)
                        task.wait()
                    end
                end)
            end

        elseif name == "Steal Floor" then
            if toggles[name] then
                -- Built-in float for Steal Floor
                local floorFloat = Instance.new("Part")
                floorFloat.Size = Vector3.new(6,1,6)
                floorFloat.Anchored = true
                floorFloat.CanCollide = true
                floorFloat.Transparency = 0.7
                floorFloat.Material = Enum.Material.ForceField
                floorFloat.Color = Color3.fromRGB(100, 255, 200)
                floorFloat.Parent = workspace

                task.spawn(function()
                    while toggles["Steal Floor"] and floorFloat and root() do
                        floorFloat.Position = root().Position - Vector3.new(0,3.2,0)
                        task.wait()
                    end
                end)

                task.spawn(function()
                    while toggles["Steal Floor"] do
                        if root() then
                            for _, part in workspace:GetPartBoundsInRadius(root().Position, 50) do
                                if part:IsA("BasePart") and part.CanCollide then
                                    part.Transparency = 0.8
                                end
                            end
                        end
                        task.wait(0.3)
                    end
                    -- Cleanup when disabled
                    if floorFloat then floorFloat:Destroy() end
                end)
            end
        end
    end)
end

makeButton("Float", 60)
makeButton("Small Hitbox", 110)
makeButton("Tween to Base", 160)
makeButton("Steal Booster", 210)
makeButton("Steal Floor", 260)

    end

    task.wait(2)
    tweenActive = false
